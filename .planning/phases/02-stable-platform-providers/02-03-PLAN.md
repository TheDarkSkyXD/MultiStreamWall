    Modify `onSearch()` in youtube.ts. Change lines 64-70 from:

    ```typescript
    const result = apiKey
      ? await this.searchViaDataApi(keyword, apiKey)
      : await this.searchViaInnertube(keyword)

    if (result.error) {

    }
    ```

    To:

    ```typescript
    let result: ProviderResult

    if (apiKey) {
      result = await this.searchViaDataApi(keyword, apiKey)
      // Fallback to Innertube on non-quota Data API errors
      // (invalid key, auth failure, server error, etc.)
      // Do NOT fallback on rate_limited (403 quota) -- that's a valid signal to stop.
      if (result.error && result.error.type !== 'rate_limited') {
        console.warn(
          `[YouTube] Data API failed for "${keyword}", falling back to Innertube: ${result.error.message}`,
        )
        result = await this.searchViaInnertube(keyword)
      }
    } else {
      result = await this.searchViaInnertube(keyword)
    }

    if (result.error) {
      return result
    }
    ```

    This ensures:
    1. Invalid API keys (400) trigger Innertube fallback -- the root cause of the UAT failure
    2. Auth errors (401) trigger fallback
    3. Server errors (5xx caught as network_error) trigger fallback
    4. Quota errors (403 rate_limited) do NOT fallback -- respects YouTube's rate limit signal
    5. No apiKey still goes directly to Innertube (unchanged)
